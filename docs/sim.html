Get scoring average for each # of dice rolls:

dice = 1:6

for (d in dice) {
  nsim = 10000
  total_score = 0
  for (i in 1:nsim) {
    rolls = sample(6, d, replace = T)
    sc = score(rolls)
    if (is.na(sc)) {
      break(paste('rolls scored as NA:', rolls))
    }
    total_score = total_score + score(rolls)
  }
  print(paste(d, 'dice roll has average score:', total_score / nsim))
}
